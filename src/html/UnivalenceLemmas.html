::

     ua-comp
       : ∀ {ℓ : Level} {A B C : Type ℓ}
       → (α : A ≃ B)
       → (β : B ≃ C)
       ---------------------------------
       → ua (α :>≃ β) == (ua α) · (ua β)

     ua-comp α β  =
      begin
        ua (α :>≃ β)
          ≡⟨ ap ua (:>≃-ite-ua α β) ⟩
        ua (ite (ua α · ua β))
          ≡⟨ ua-η ((ua α) · (ua β)) ⟩
        (ua α) · (ua β)
      ∎

::

     ua-inv
       : ∀ {ℓ : Level} {A B : Type ℓ}
       → (α : A ≃ B)
       -------------------------
       → ua (≃-sym α) ≡ ! (ua α)

     ua-inv α =
       begin
         ua (≃-sym α)
           ≡⟨ ap (_· ua (≃-sym α)) (! (·-linv (ua α))) ⟩
         ! (ua α) · ua α · ua (≃-sym α)
           ≡⟨ ·-assoc (! (ua α)) _ _ ⟩
         ! (ua α) · (ua α · ua (≃-sym α))
           ≡⟨ ap (! (ua α) ·_) (ua-inv-r α) ⟩
         ! (ua α) · refl _
           ≡⟨ ! (·-runit (! ((ua α)))) ⟩
         ! (ua α)
       ∎
